# Função para converter uma sequência de DNA em RNA
def dna_para_rna(dna):
    # Substitui "T" por "U" para gerar a sequência de RNA
    rna = dna.upper().replace("T", "U")
    return rna

# Função para exibir o DNA e RNA lado a lado
def exibir_dna_rna(dna):
    rna = dna_para_rna(dna)
    print("DNA e RNA transcritos lado a lado:")
    print("DNA: " + dna)
    print("RNA: " + rna)
